# 25、请编写一个 JavaScript 函数 parseQueryString，它的用途是把 URL 参数解析为一个对象

/*！
* @param { String } url 需要解析的地址
*/
function parseQueryString(url){
    // url 不存在
    if (!url){
      return {};
    }

    // 正则
    // 匹配开头不为?或者=或者&或者#的至少一个字符，以=连接，=后为非=或者&或者#至少一个字符。
    var reg = /([^?=&#]+)=([^=&#]+)/g;
    var result = '';
    var query = {};

    // reg必须以变量的形式传入，否则会陷入死循环
    while((result = reg.exec(url))!== null){
      query[result[1]] = result[2];
    }
    return query;
}
